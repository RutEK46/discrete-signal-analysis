    e^{ix} = \sum_{n=0}^\infty \frac{(ix)^n}{n!}
\end{equation}

\begin{equation}
    e^{ix} = \frac{(ix)^0}{0!} + \frac{(ix)^1}{1!} + \frac{(ix)^2}{2!} + \frac{(ix)^3}{3!} + \frac{(ix)^4}{4!} + \frac{(ix)^5}{5!} + ...
